
#include <stdio.h>


 
void lpyr()
{
    int year;
   printf("Enter a year: ");
   scanf("%d", &year);


   if (year % 400 == 0) {
      printf("%d is a leap year.", year);
   }
  
   else if (year % 100 == 0) {
      printf("%d is not a leap year.", year);
   }
   
   else if (year % 4 == 0) {
      printf("%d is a leap year.", year);
   }
 
   else {
      printf("%d is not a leap year.", year);
   }

   

}
void main() 
{  
   
   lpyr(); 
   
}

OUTPUT:-
 
Enter a year: 2020
2020 is a leap year.
